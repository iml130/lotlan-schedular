# Basic Template structure
template Palette
	type
	value # readonly
end

template PaletteSensor # A Sensor which sets the Value to true if full
	value # readonly
end

template Container
    type
end

Container container
    type = "Container"
end

Container room
    type = "Supply_Room"
end



# Basic Instance structure
Palette palette_1
	type  =	LightSensor
	value = "boolean"
end

Palette palette_2
	type  =	LightSensor
	value = "boolean"
end

Palette palette_3
	type  =	LightSensor
	value = "boolean"
end

PaletteSensor palette_1Full
	value = "Boolean"
end

PaletteSensor palette_2Full
	value = "Boolean"
end

PaletteSensor palette_3Full
	value = "Boolean"
end


task Reset_Task
	Transport 
    from 		palette_1, palette_2, palette_3 
    to 			room
end 


# Basic Task structure
task Transport_Task
    # Transport something from A to B
	Transport 
    from 		palette_1, palette_2, palette_3 
    to 			container
    
	TriggeredBy	palette_1Full.value == TRUE
	TriggeredBy	palette_2Full.value == FALSE
	TriggeredBy	palette_3Full.value == !12 || 42.42 <= 42.31
    
	OnDone		Reset_Task
end
