# Basic Task structure
task Transport_Task
    # Transport something from A to B
	Transport 
    from 		palette_1, palette_2, palette_3 
    to 			container
    
	TriggeredBy	palette_1Full
	TriggeredBy	palette_2Full
	TriggeredBy	palette_3Full
    
	OnDone		Reset_Task
end
